now lets plot gross horsepower on x axis and weight on y axis

mtcars %>%
  ggplot()+
  aes(x=hp,y=wt)+
  geom_point()+
  xlab("gross horsepower")+
  ylab("weight")+
  ggtitle("Scatter plot") +
  theme_bw()
