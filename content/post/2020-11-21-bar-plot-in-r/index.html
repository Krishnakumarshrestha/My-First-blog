1 lets plot

plot1<-chickwts %>% 
  count(feed) %>%
  ggplot() +
  aes(x=feed,weights=n)+
  geom_bar(aes(fill=feed))+
  ylim(c(0,20))+
  geom_text(aes(x=feed,y=n,label=paste0("N=",n)),vjust=-0.3)+
  labs(
    title = "Bar plot",
    subtitle = paste0("ggplot2")
  )+theme_economist()
plot1

plotly::ggplotly(plot1)
